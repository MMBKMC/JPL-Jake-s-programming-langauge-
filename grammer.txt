<program> ::= <statement_list>

<statement_list> ::= <statement> <statement_list> | <statement>

<statement> ::= <display_statement>
              | <define_statement>
              | <input_statement>
              | <if_statement>
              | <while_statement>
              | <end_statement>

<display_statement> ::= "DISPLAY" <expression>

<define_statement> ::= "DEFINE" <identifier> <expression>

<input_statement> ::= "INPUT" <string>

<if_statement> ::= "IF" <expression> "THEN" <statement_list> <else_clause> "ENDIF"
<else_clause> ::= "ELSE" <statement_list> | ε

<while_statement> ::= "WHILE" <expression> "DO" <statement_list> "ENDWHILE"

<end_statement> ::= "STOP" "RUN" | ε

<expression> ::= <term> <expression_prime>
<expression_prime> ::= "+" <term> <expression_prime> | ε
<term> ::= <factor> <term_prime>
<term_prime> ::= "*" <factor> <term_prime> | ε
<factor> ::= <identifier> | <number> | <string> | "(" <expression> ")"

<identifier> ::= [a-zA-Z_][a-zA-Z0-9_]*
<number> ::= [0-9]+
<string> ::= "\"" .*? "\""
